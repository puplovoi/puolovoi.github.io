#include <stdio.h>
#include <stdlib.h>

// 定义单链表节点结构体
typedef struct LNode {
    char data;
    struct LNode* next;
} LinkNode;
// 初始化链表
 void InitList(LinkNode *&L);
// 尾插法插入节点
 void Insert(LinkNode *&L,char x);
 // 打印链表
 void PrintList(LinkNode *L);
 // 销毁链表
 void DestroyList(LinkNode *&L);
 // 输出单链表的长度
 int Length(LinkNode *L);
//输出单链表第x个元素
void GetElem(LinkNode *L,int x);
//输出某个元素的位置
void LocateElem(LinkNode *L,char x);
//尾插法在第a个位置插入元素x
void InsertElem(LinkNode *&L,char x,int a);
 // 输出单链表是否为空
 int IsEmpty(LinkNode *L);
//删除链表第a个元素
void DeleteElem(LinkNode *&L,int a);

int main()
{
    LinkNode *L=NULL;
    InitList(L);//初始化链表
    Insert(L,'a');
    Insert(L,'b');
    Insert(L,'c');
    Insert(L,'d');
    Insert(L,'e');
    printf("链表插入元素后:\n");
    PrintList(L);
    printf("链表的长度为:%d\n",Length(L)); //输出链表长度
    if(IsEmpty(L))    //链表为空
    {
        printf("链表为空!\n");
    }
    else    //链表不为空
    {
        printf("链表不为空!\n");
    }
    LocateElem(L,'c');    //输出c元素位置
    GetElem(L,3);    //输出第3个元素
    InsertElem(L,'f',4); //在第4个位置插入元素f
    printf("插入元素f后:\n");
    PrintList(L);
    DeleteElem(L,2); //删除第2个元素
    DestroyList(L);
    return 0;
}

 void InitList(LinkNode *&L)
 {
    L=(LinkNode *)malloc(sizeof(LinkNode));    //创建头结点
    L->next=NULL;
 }

  void Insert(LinkNode *&L,char x)
 {
    LinkNode *p=(LinkNode *)malloc(sizeof(LinkNode));
    p->data=x;
    p->next=NULL;
    LinkNode *q=L;
    while(q->next!=NULL)    //找到尾节点
    {
        q=q->next;
    }
    q->next=p;    //插入到尾节点后
 }

  void PrintList(LinkNode *L)
 {
    if(L->next==NULL)    //链表为空
    {
        printf("链表为空!\n");
        return;
    }
    LinkNode *p=L->next;
    while(p!=NULL)    //遍历链表
    {
        printf("%c ",p->data);
        p=p->next;
    }
    printf("\n");
 }

  void DestroyList(LinkNode *&L)
 {
    LinkNode *p=L->next;
    LinkNode *q;
    while(p!=NULL)    //遍历链表
    {
        q=p;
        p=p->next;
        free(q);    //释放节点
    }
    free(L);    //释放头结点
    L=NULL;
 }

 int Length(LinkNode *L)
 {
    int len=0;
    LinkNode *p=L->next;
    while(p!=NULL)    //遍历链表
    {
        len++;
        p=p->next;
    }
    return len;
 } 

void GetElem(LinkNode *L,int x)
{
    int i=1;
    LinkNode *p=L->next;
    while(p!=NULL)    //遍历链表
    {
        if(i==x)    //找到第x个元素
        {
            printf("%d的元素是:%c\n", x, p->data); //输出元素
            return;
        }
        i++;
        p=p->next;
    }
    printf("未找到第%d个元素!\n", x);
    return;    //未找到第x个元素
}

void LocateElem(LinkNode *L,char x)
{
    int i=0;
    LinkNode *p=L;
    while(p!=NULL)    //遍历链表
    {
        if(p->data==x)    //找到元素
        {
            printf("%c的位置是:%d\n", x, i); //输出元素位置
            return;
        }
        i++;
        p=p->next;
    }
    printf("%c元素不存在!\n", x);
    return;
}

void InsertElem(LinkNode *&L,char x,int a)
{
    LinkNode *p = L;
    int i = 1;
    while (p!=NULL && i<a)    //找到第a-1个元素
    {
        p=p->next;
        i++;
    }
     if (p==NULL)    //链表长度小于a，插入到尾节点后   
     {
        Insert(L,x);
        return;
     }
     LinkNode *q=(LinkNode *)malloc(sizeof(LinkNode));    //创建新节点
     q->data=x;
     q->next=p->next;
     p->next=q;
}

 int IsEmpty(LinkNode *L)
 {
    return L->next==NULL;
 }

 void DeleteElem(LinkNode *&L,int a)
{
    LinkNode *p=L;
    int i=1;
    while(p!=NULL && i<a)    //找到第a-1个元素
    {
        p=p->next;
        i++;
    }
    if(p==NULL)    //链表长度小于a，不删除
    {
        printf("链表长度小于%d，不删除!\n", a);
        printf("链表未改变:\n");
        PrintList(L);
        return;
    }
    LinkNode *q=p->next;
    p->next=q->next;
    free(q);    //释放节点
    printf("删除第%d个元素后:\n", a);
    PrintList(L);
}
